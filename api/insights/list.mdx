# List Insights

Retrieve a list of all insights (saved charts and analyses) in your PostHog project. Supports filtering by type, dashboard, tags, and more.

## Endpoint

```http
GET /api/projects/:project_id/insights/
```

## Path Parameters

<ParamField path="project_id" type="integer" required>
  The ID of the project to retrieve insights from
</ParamField>

## Query Parameters

<ParamField query="saved" type="boolean">
  Filter by saved status:
  - `true` - Only saved insights (or insights on dashboards)
  - `false` - Only unsaved/temporary insights
</ParamField>

<ParamField query="insight" type="string">
  Filter by insight type:
  - `TRENDS` - Trends analysis
  - `FUNNELS` - Conversion funnels
  - `RETENTION` - Retention analysis
  - `PATHS` - User path analysis
  - `STICKINESS` - Stickiness/engagement
  - `LIFECYCLE` - Lifecycle analysis
  - `SQL` - SQL/HogQL queries
  - `JSON` - Custom JSON insights
</ParamField>

<ParamField query="search" type="string">
  Search insights by name, description, or tags (case-insensitive)
</ParamField>

<ParamField query="dashboards" type="string">
  JSON array of dashboard IDs to filter by. Returns insights on ANY of the specified dashboards.
  
  Example: `[123, 456]`
</ParamField>

<ParamField query="tags" type="string">
  JSON array of tag names to filter by
  
  Example: `["marketing", "product"]`
</ParamField>

<ParamField query="user" type="boolean">
  If `true`, filter to insights created by the current user
</ParamField>

<ParamField query="favorited" type="boolean">
  If `true`, filter to favorited insights only
</ParamField>

<ParamField query="created_by" type="string">
  JSON array of user IDs to filter by creator
  
  Example: `[123, 456]`
</ParamField>

<ParamField query="date_from" type="string">
  Filter insights modified after this date (relative date format like `-7d` or ISO 8601)
</ParamField>

<ParamField query="date_to" type="string">
  Filter insights modified before this date
</ParamField>

<ParamField query="order" type="string">
  Sort order:
  - `order` (default) - Manual ordering
  - `-last_viewed_at` - Most recently viewed first
  - `last_viewed_at` - Least recently viewed first
</ParamField>

<ParamField query="limit" type="integer" default={100}>
  Maximum number of results to return
</ParamField>

<ParamField query="offset" type="integer" default={0}>
  Number of results to skip (for pagination)
</ParamField>

<ParamField query="basic" type="boolean" default={false}>
  If `true`, return basic metadata only (no query results, faster response)
</ParamField>

<ParamField query="refresh" type="string">
  How to handle result calculation:
  - `force_cache` - Return cached data only
  - `blocking` - Calculate synchronously if needed
  - `async` - Kick off background calculation if needed
  - `lazy_async` - Background calculation for stale cache
  - `force_blocking` - Always calculate synchronously
  - `force_async` - Always kick off background calculation
</ParamField>

## Response Fields

<ResponseField name="results" type="array">
  Array of insight objects
  
  <ResponseField name="id" type="integer">
    Unique identifier for the insight
  </ResponseField>
  
  <ResponseField name="short_id" type="string">
    Short, human-readable ID (used in URLs)
  </ResponseField>
  
  <ResponseField name="name" type="string">
    Name of the insight
  </ResponseField>
  
  <ResponseField name="derived_name" type="string">
    Auto-generated name if no explicit name provided
  </ResponseField>
  
  <ResponseField name="description" type="string">
    Description of the insight
  </ResponseField>
  
  <ResponseField name="query" type="object">
    Query definition for this insight
  </ResponseField>
  
  <ResponseField name="result" type="any">
    Query results (if not using `basic=true`)
  </ResponseField>
  
  <ResponseField name="created_by" type="object">
    User who created the insight
  </ResponseField>
  
  <ResponseField name="created_at" type="string">
    ISO 8601 timestamp of creation
  </ResponseField>
  
  <ResponseField name="last_modified_at" type="string">
    ISO 8601 timestamp of last modification
  </ResponseField>
  
  <ResponseField name="last_refresh" type="string">
    ISO 8601 timestamp of when results were calculated
  </ResponseField>
  
  <ResponseField name="saved" type="boolean">
    Whether this insight is saved
  </ResponseField>
  
  <ResponseField name="favorited" type="boolean">
    Whether this insight is favorited
  </ResponseField>
  
  <ResponseField name="dashboards" type="array">
    Array of dashboard IDs this insight is on
  </ResponseField>
  
  <ResponseField name="tags" type="array">
    Array of tags applied to this insight
  </ResponseField>
</ResponseField>

<ResponseField name="count" type="integer">
  Total number of insights matching filters
</ResponseField>

<ResponseField name="next" type="string">
  URL for next page of results
</ResponseField>

<ResponseField name="previous" type="string">
  URL for previous page of results
</ResponseField>

## Examples

### List All Insights

Retrieve all saved insights with basic metadata:

<CodeGroup>
```bash cURL
curl -X GET "https://app.posthog.com/api/projects/12345/insights/?basic=true&saved=true" \
  -H "Authorization: Bearer YOUR_API_KEY"
```

```python Python
import posthog

posthog.project_api_key = 'YOUR_API_KEY'

insights = posthog.get('/api/projects/12345/insights/', params={
    'basic': True,
    'saved': True
})

print(f"Found {insights['count']} insights")
for insight in insights['results']:
    print(f"- {insight['name']} (ID: {insight['short_id']})")
```

```javascript JavaScript
const response = await fetch(
  'https://app.posthog.com/api/projects/12345/insights/?basic=true&saved=true',
  {
    headers: {
      'Authorization': 'Bearer YOUR_API_KEY'
    }
  }
);

const data = await response.json();
console.log(`Found ${data.count} insights`);
data.results.forEach(insight => {
  console.log(`- ${insight.name} (ID: ${insight.short_id})`);
});
```
</CodeGroup>

### Filter by Dashboard

Get insights on specific dashboards:

<CodeGroup>
```bash cURL
curl -X GET "https://app.posthog.com/api/projects/12345/insights/?dashboards=%5B456%2C789%5D" \
  -H "Authorization: Bearer YOUR_API_KEY"
```

```python Python
import posthog
import json

posthog.project_api_key = 'YOUR_API_KEY'

insights = posthog.get('/api/projects/12345/insights/', params={
    'dashboards': json.dumps([456, 789])
})

print(f"Found {insights['count']} insights on these dashboards")
```

```javascript JavaScript
const dashboardIds = [456, 789];
const response = await fetch(
  `https://app.posthog.com/api/projects/12345/insights/?dashboards=${encodeURIComponent(JSON.stringify(dashboardIds))}`,
  {
    headers: {
      'Authorization': 'Bearer YOUR_API_KEY'
    }
  }
);

const data = await response.json();
console.log(`Found ${data.count} insights on these dashboards`);
```
</CodeGroup>

### Recently Viewed Insights

Get insights sorted by most recently viewed:

<CodeGroup>
```bash cURL
curl -X GET "https://app.posthog.com/api/projects/12345/insights/?order=-last_viewed_at&limit=10" \
  -H "Authorization: Bearer YOUR_API_KEY"
```

```python Python
import posthog

posthog.project_api_key = 'YOUR_API_KEY'

recent = posthog.get('/api/projects/12345/insights/', params={
    'order': '-last_viewed_at',
    'limit': 10
})

for insight in recent['results']:
    print(f"{insight['name']} - Last viewed: {insight.get('last_viewed_at', 'Never')}")
```

```javascript JavaScript
const response = await fetch(
  'https://app.posthog.com/api/projects/12345/insights/?order=-last_viewed_at&limit=10',
  {
    headers: {
      'Authorization': 'Bearer YOUR_API_KEY'
    }
  }
);

const data = await response.json();
data.results.forEach(insight => {
  console.log(`${insight.name} - Last viewed: ${insight.last_viewed_at || 'Never'}`);
});
```
</CodeGroup>

## Notes

- Use `basic=true` for faster responses when you only need metadata
- Results are paginated with a default limit of 100
- Deleted insights are excluded from results
- The `refresh` parameter controls whether and how to recalculate results
- Use `force_cache` for fastest responses (may return stale data)
- Multiple filters are combined with AND logic
- Search is case-insensitive and matches partial strings
